function convertHTML(str) {
  // & = &amp; < = &lt; > = &gt; " = &quot; ' = &apos;
  var spltStr = str.split('');

  for (var i = 0; i < spltStr.length; i++) {
    switch (spltStr[i]) {
      case '&':
        spltStr[i] = '&amp;';
        break;
      case '<':
        spltStr[i] = '&lt;';
        break;
      case '>':
        spltStr[i] = '&gt;';
        break;
      case '"':
        spltStr[i] = '&quot;';
        break;
      case "'":
        spltStr[i] = "&apos;";
        break;
    }
  }
  return spltStr.join('');
}

    let ln = 0, cn = 1, rs = 0;
